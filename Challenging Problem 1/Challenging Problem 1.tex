\section*{Question: Challenging Problem 1}
Let X be a random variable such that $E(X)=E(X^2)=1$. Then $E(X^{100})=?$
\begin{enumerate}[label=(\Alph*)]
    \item 0
    \item 1
    \item $2^{100}$
    \item $2^{100}+1$
\end{enumerate}
\section*{Solution}
\begin{align}
    \rho^2&=\mean{X^2}-\mean{X}^2\\
          &=1 - 1^2\\
          &=0
\end{align}
$\therefore$ X is constant and equal to 1


Thus we can conclude that
\begin{align}
    \mean{X^{100}}&=\sum X^{100}\pr{X}\\
                  &=1
\end{align}
\solution{Option B}
